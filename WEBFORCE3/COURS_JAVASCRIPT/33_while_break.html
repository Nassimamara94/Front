EXERCICE avec La boucle while et break

break met fin à la boucle 

         var prenoms = '', prenom, proceed = true;
         while (proceed) { 
             prenom = prompt("saisissez un prénom ?");
             if (prenom) { 
                 prenoms += prenom + ' ';
                } else { 
                    proceed = break; 
                 }
                 } // fin de la boucle while 
                 
    document.write(prenoms);
